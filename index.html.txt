function navCalc(){
  if(!navState.mat) return;
  const depth=parseFloat(document.getElementById('nav-depth').value);
  const liftoff=parseFloat(document.getElementById('nav-liftoff').value);
  let m={...navState.mat};
  if(navState.magType==='mag' && navState.sat==='sat') m.mur=Math.max(1,m.mur/100);

  const d_m=depth/1000;
  const f_opt=Math.pow(503/d_m,2)/(m.mur*m.sigma);
  const f_lo =Math.pow(503/(d_m*2),2)/(m.mur*m.sigma);
  const f_hi =Math.pow(503/(d_m*0.5),2)/(m.mur*m.sigma);
  const fok=fmtF(f_opt/1000), flo=fmtF(f_lo/1000), fhi=fmtF(f_hi/1000);

  document.getElementById('nv-fopt').textContent=fok.v;
  document.getElementById('nv-fopt-u').textContent=fok.u;
  document.getElementById('nv-flo').textContent=flo.v;
  document.getElementById('nv-flo-u').textContent=flo.u;
  document.getElementById('nv-fhi').textContent=fhi.v;
  document.getElementById('nv-fhi-u').textContent=fhi.u;

  // バー
  const lMin=Math.log10(1),lMax=Math.log10(10000);
  const lLo=Math.log10(Math.max(1,f_lo/1000)), lHi=Math.log10(Math.min(10000,f_hi/1000));
  const bl=((lLo-lMin)/(lMax-lMin)*100).toFixed(1);
  const bw=((lHi-lLo)/(lMax-lMin)*100).toFixed(1);
  document.getElementById('nv-bar').style.cssText=`left:${bl}%;width:${bw}%`;
  document.getElementById('nv-bar-label').textContent=`推奨帯域: ${flo.v}${flo.u} 〜 ${fhi.v}${fhi.u}`;

  // 浸透深さテーブル
  const freqs=[1,2,5,10,20,50,100,200,500,1000,2000,5000,10000];
  let tbody='';
  freqs.forEach(fk=>{
    const dv=delta_mm(fk*1000,m.mur,m.sigma);
    const inRange=(fk*1000>=f_lo*0.7 && fk*1000<=f_hi*1.3);
    const judge=dv>=depth*0.8&&dv<=depth*3?'✅':'—';
    const ratio=dv>=depth?(dv/depth).toFixed(1)+'倍':(dv/depth*100).toFixed(0)+'%';
    tbody+=`<tr class="${inRange?'hl':''}"><td>${fk>=1000?(fk/1000)+'MHz':fk+'kHz'}</td><td class="d-val">${dv.toFixed(3)} mm</td><td>${ratio}</td><td>${judge}</td></tr>`;
  });
  document.getElementById('nv-depth-body').innerHTML=tbody;

  // フェライト推奨
  const fopt_MHz=f_opt/1e6;
  const recs=FERRITES.filter(f=>f.fmax_MHz>=fopt_MHz*0.5).slice(0,4);
  document.getElementById('nv-ferrite').innerHTML=recs.map(f=>{
    const [sl,sc]=SUIT_LABEL[f.suit]||['—',''];
    return `<div class="ferrite-card ${f.type}">
      <div class="ferrite-name ${f.type}${f.cur?' cur':''}">${f.cur?'◆':f.suit==='best'?'★':''}${f.name}</div>
      <div class="ferrite-info"><div class="ferrite-freq">${f.type==='mnzn'?'MnZn':'NiZn'}　μ=${f.mu.toLocaleString()}　〜${f.fmax_MHz>=1?f.fmax_MHz+'MHz':f.fmax_MHz*1000+'kHz'}</div><div class="ferrite-note">${f.note}</div></div>
      <div class="suit-badge ${sc}">${sl}</div>
    </div>`;
  }).join('');

  // コイル推奨値
  // 推奨線径 = 表皮深さ×2（実際の材質の導電率を使用）
  // ただし線材は銅線なので銅の表皮深さで計算
  
  // 銅線の表皮深さ（mm）= 0.066/√(f_Hz) * 1000
  const skinDepth_mm = 66.04/Math.sqrt(f_opt); // 銅線表皮深さ(mm) δ=66.04/√f_Hz

  document.getElementById('nv-rc-dz').textContent=f_opt>500000?200:100;

  // 磁性体警告
  const warnEl=document.getElementById('nav-mag-warn');
  if(navState.magType==='mag'){
    warnEl.style.display='block';
    warnEl.innerHTML=navState.sat==='sat'
      ?'<strong>🧲 磁気飽和あり：</strong>永久磁石でμrを約1/100に低減。非磁性体に近い状態で探傷できます。'

  } else { warnEl.style.display='none'; }

     <strong>次のステップ：</strong>「円筒形」または「長方形」タブでコイルを設計してください。`;

  document.getElementById('nav-result').style.display='block';
  document.getElementById('nav-result').className='fade-in';

function calcCyl(){
  const id=+document.getElementById('c-id').value, od=+document.getElementById('c-od').value;
  const len=+document.getElementById('c-len').value, wd=+document.getElementById('c-wd').value;
  const N=+document.getElementById('c-turns').value;
  const fs=[+document.getElementById('c-f1').value,+document.getElementById('c-f2').value,+document.getElementById('c-f3').value];
  const wh=(od-id)/2, ml=Math.floor(wh/wd), pl=Math.floor(len/wd), mt=ml*pl;
  const avgR=(id/2+od/2)/2/1000, area=Math.PI*avgR*avgR;
  const coilOD=id+2*ml*wd, L=MU0*N*N*area/(len/1000);
  document.getElementById('cr-wh').textContent=f2(wh);
  document.getElementById('cr-ml').textContent=ml;
  document.getElementById('cr-pl').textContent=pl;
  document.getElementById('cr-mt').textContent=mt.toLocaleString();
  document.getElementById('cr-od').textContent=f2(coilOD);
  document.getElementById('cr-l').textContent=f3(L*1e6);
  const w=document.getElementById('cr-warn');
  if(N>mt){w.style.display='block';w.textContent='⚠️ 巻き数が最大値を超えています！';}else{w.style.display='none';}
  let fb='',ib='';
  fs.forEach(f=>{
    const Z=2*Math.PI*f*1000*L, d=503/Math.sqrt(f*1000*1*5.8e7)*1000;
    const lv=Z<50?'📉低い':Z>500?'📈高い':'✅良好';
    const jd=wd<=d*2?'✅':'⚠️表皮';
    fb+=`<tr><td>${f>=1000?f/1000+'MHz':f+'kHz'}</td><td class="z-val">${f1(Z)}</td><td class="d-val">${f3(d)}mm</td><td>${jd}</td></tr>`;
    const Nt=Math.round(Math.sqrt(100*(len/1000)/(2*Math.PI*f*1000*MU0*area)));
    const ok=Nt<=mt, ly=Math.ceil(Nt/pl);
    ib+=`<tr><td>${f>=1000?f/1000+'MHz':f+'kHz'}</td><td class="z-val">${Nt.toLocaleString()}T</td><td>${ok?'✅':'❌'}</td><td>${ly}層</td></tr>`;
  });
  document.getElementById('cr-freq').innerHTML=fb;
  document.getElementById('cr-inv').innerHTML=ib;
  document.getElementById('cyl-res').style.display='block';
}

function calcRect(){
  const ih=+document.getElementById('r-ih').value, iw=+document.getElementById('r-iw').value;
  const oh=+document.getElementById('r-oh').value, ow=+document.getElementById('r-ow').value;
  const len=+document.getElementById('r-len').value, wd=+document.getElementById('r-wd').value;
  const N=+document.getElementById('r-turns').value;
  const fs=[+document.getElementById('r-f1').value,+document.getElementById('r-f2').value,+document.getElementById('r-f3').value];
  const eff=Math.min((oh-ih)/2,(ow-iw)/2), ml=Math.floor(eff/wd), pl=Math.floor(len/wd), mt=ml*pl;
  const area=((ih+oh)/2/1000)*((iw+ow)/2/1000);
  const L=MU0*N*N*area/(len/1000);
  document.getElementById('rr-wh').textContent=f2(eff);
  document.getElementById('rr-ml').textContent=ml;
  document.getElementById('rr-pl').textContent=pl;
  document.getElementById('rr-mt').textContent=mt.toLocaleString();
  document.getElementById('rr-size').textContent=f2(ih+2*ml*wd)+'×'+f2(iw+2*ml*wd);
  document.getElementById('rr-l').textContent=f3(L*1e6);
  let fb='';
  fs.forEach(f=>{
    const Z=2*Math.PI*f*1000*L, d=503/Math.sqrt(f*1000*1*5.8e7)*1000;
    const jd=wd<=d*2?'✅':'⚠️表皮';
    fb+=`<tr><td>${f>=1000?f/1000+'MHz':f+'kHz'}</td><td class="z-val">${f1(Z)}</td><td class="d-val">${f3(d)}mm</td><td>${jd}</td></tr>`;
  });
  document.getElementById('rr-freq').innerHTML=fb;
  document.getElementById('rect-res').style.display='block';
}

function calcInv(){
  const N1=+document.getElementById('i-n1').value, Z1=+document.getElementById('i-z1').value;
  const f=+document.getElementById('i-f').value, Zt=+document.getElementById('i-zt').value;
  const tol=+document.getElementById('i-tol').value;
  const L1=Z1/(2*Math.PI*f*1000)*1e6, Lt=Zt/(2*Math.PI*f*1000)*1e6;
  const Nt=Math.round(N1*Math.sqrt(Zt/Z1)), Nu=Math.ceil(N1*Math.sqrt(Zt/Z1)), Nd=Math.floor(N1*Math.sqrt(Zt/Z1));
  const Zv=Z1*(Nt/N1)**2, err=Math.abs(Zv-Zt)/Zt*100;
  document.getElementById('ir-n').textContent=Nt.toLocaleString();
  document.getElementById('ir-zv').textContent=f1(Zv);
  document.getElementById('ir-err').textContent=f1(err);
  document.getElementById('ir-l1').textContent=f3(L1);
  document.getElementById('ir-lt').textContent=f3(Lt);
  document.getElementById('ir-nu').textContent=Nu.toLocaleString();
  document.getElementById('ir-nd').textContent=Nd.toLocaleString();
  let rows='';
  for(let i=-10;i<=10;i++){
    const n=Math.max(1,Nt+i), z=Z1*(n/N1)**2, e=Math.abs(z-Zt)/Zt*100, ok=e<=tol;
    const dir=z>Zt?'▲':'▼';
    rows+=`<tr class="${i===0?'hl':''}"><td>${n.toLocaleString()}</td><td>${f1(z)}</td><td>${f1(e)}</td><td>${ok?'<span style="color:var(--accent2)">✅</span>':dir}</td></tr>`;
  }
  document.getElementById('ir-table').innerHTML=rows;
  document.getElementById('inv-res').style.display='block';
}

}
